Var
// Seção de Declarações das variáveis
   n1,n2,r:real
   operacao:caracter


Inicio
// Seção de Comandos, procedimento, funções, operadores, etc... 
   escreva("Digite o primeiro número: ")
   leia(n1)
   escreva("Digite a operação( + - * / ): ")
   leia(operacao)
   escreva("Digite o segundo número: ")
   leia(n2)

     escolha operacao
             caso "+"
                  r <- n1 + n2
             caso "-"
                  r <- n1 - n2
             caso "*"
                  r <- n1 * n2
             caso "/"
                  r <- n1 / n2
     fimescolha
     
     limpatela()
     escreva(n1,"",operacao,"",n2," = ",r)
